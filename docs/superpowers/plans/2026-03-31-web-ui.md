> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a browser-based PWA (Nuxt 4 + Nuxt UI 4) hosted on GitHub Pages that manages ESP32 devices running ZenClaw — provisioning via Web Serial and daily management via HTTPS/WebSocket.

**Architecture:** Monorepo with `firmware/` (all ESP32 code) and `web/` (Nuxt app) as siblings. The ESP32 runs a microdot HTTPS server exposing REST + WebSocket APIs. The webapp connects via Web Serial (USB) for provisioning and HTTPS for network management.

**Tech Stack:** Nuxt 4, Nuxt UI 4, TypeScript, esptool-js, Web Serial API, microdot (MicroPython), mbedTLS/SSL

**Spec:** `docs/superpowers/specs/2026-03-31-web-ui-design.md`

```python
"""
ZenClaw API server — microdot HTTPS with CORS for the web UI.
Runs as an async background task alongside Telegram and heartbeat.
"""

from lib.microdot import Microdot
from lib.microdot.cors import CORS
from lib.sys.log import log

app = Microdot()
CORS(app, allowed_origins='*', allow_credentials=False)


def _get_ssl_context():
    """Load or generate self-signed SSL cert. Returns ssl context or None."""
    import os
    import ssl

    cert_dir = 'data/certs'
    cert_path = '{}/cert.pem'.format(cert_dir)
    key_path = '{}/key.pem'.format(cert_dir)

    try:
        os.stat(cert_path)
        os.stat(key_path)
    except OSError:
        log('info', 'API: generating self-signed cert', source='api')
        try:
            os.mkdir(cert_dir)
        except:
            pass
        _generate_self_signed_cert(cert_path, key_path)

    ctx = ssl.SSLContext(ssl.PROTOCOL_TLS_SERVER)
    ctx.load_cert_chain(cert_path, key_path)
    return ctx


def _generate_self_signed_cert(cert_path, key_path):
    """Generate a self-signed cert using MicroPython's built-in crypto."""
    # MicroPython ESP32 doesn't have a cert generation API in stdlib.
    # We ship a pre-generated cert pair in firmware, or generate on first
    # boot using a minimal ASN.1 builder.
    # For now, copy the bundled default cert.
    import os
    src_cert = 'lib/api/default_cert.pem'
    src_key = 'lib/api/default_key.pem'
    try:
        os.stat(src_cert)
        with open(src_cert, 'rb') as f:
            data = f.read()
        with open(cert_path, 'wb') as f:
            f.write(data)
        with open(src_key, 'rb') as f:
            data = f.read()
        with open(key_path, 'wb') as f:
            f.write(data)
        log('info', 'API: copied default cert to {}'.format(cert_path), source='api')
    except Exception as e:
        log('error', 'API: cert generation failed: {}'.format(e), source='api')
        raise


async def start(port=8443):
    """Start the API server as a background task."""
    ssl_ctx = _get_ssl_context()
    log('info', 'API: starting HTTPS server on port {}'.format(port), source='api')
    await app.start_server(port=port, ssl=ssl_ctx)
```

```python
"""System status API endpoints."""

import gc
from lib.api.server import app
from lib.sys.log import log


@app.get('/api/status')
async def get_status(req):
    gc.collect()
    info = {
        'agent_name': 'ZenClaw',
        'version': _get_version(),
    }

    # Memory
    try:
        free = gc.mem_free()
        alloc = gc.mem_alloc()
        info['memory'] = {
            'free_kb': free // 1024,
            'used_kb': alloc // 1024,
            'total_kb': (free + alloc) // 1024,
        }
    except:
        info['memory'] = None

    # CPU temperature
    try:
        import esp32
        info['temperature_c'] = esp32.mcu_temperature()
    except:
        info['temperature_c'] = None

    # WiFi
    try:
        from lib.wifi import is_connected, get_ip
        import network
        wlan = network.WLAN(network.STA_IF)
        info['wifi'] = {
            'connected': is_connected(),
            'ip': get_ip(),
            'rssi': wlan.status('rssi') if is_connected() else None,
        }
    except:
        info['wifi'] = None

    # Uptime
    try:
        import time
        info['uptime_s'] = time.ticks_ms() // 1000
    except:
        try:
            import time
            info['uptime_s'] = int(time.time())
        except:
            info['uptime_s'] = None

    # Storage
    try:
        import os
        stat = os.statvfs('/')
        block_size = stat[0]
        total_blocks = stat[2]
        free_blocks = stat[3]
        info['storage'] = {
            'total_kb': (block_size * total_blocks) // 1024,
            'free_kb': (block_size * free_blocks) // 1024,
        }
    except:
        info['storage'] = None

    return info


@app.post('/api/restart')
async def restart(req):
    log('info', 'API: restart requested', source='api')
    try:
        import machine
        machine.reset()
    except ImportError:
        return {'error': 'restart not available on desktop'}, 501
    return {'ok': True}


def _get_version():
    try:
        import json
        with open('firmware-version.json') as f:
            return json.load(f).get('platform', 'unknown')
    except:
        return 'unknown'
```

```python
"""File management API endpoints."""

import os
from lib.api.server import app


@app.get('/api/files')
async def list_dir(req):
    path = req.args.get('path', '/')
    try:
        entries = []
        for name in os.listdir(path):
            full = '{}/{}'.format(path.rstrip('/'), name)
            try:
                stat = os.stat(full)
                is_dir = stat[0] & 0x4000 != 0
                entries.append({
                    'name': name,
                    'path': full,
                    'is_dir': is_dir,
                    'size': stat[6] if not is_dir else None,
                })
            except:
                entries.append({'name': name, 'path': full, 'is_dir': False, 'size': None})
        return {'path': path, 'entries': entries}
    except OSError as e:
        return {'error': 'Cannot list {}: {}'.format(path, e)}, 404


@app.get('/api/files/read')
async def read_file(req):
    path = req.args.get('path', '')
    if not path:
        return {'error': 'path required'}, 400
    try:
        with open(path, 'r') as f:
            content = f.read()
        return {'path': path, 'content': content}
    except OSError:
        return {'error': 'File not found: {}'.format(path)}, 404


@app.put('/api/files/write')
async def write_file(req):
    body = req.json
    path = body.get('path', '')
    content = body.get('content', '')
    if not path:
        return {'error': 'path required'}, 400
    # Ensure parent directory exists
    parts = path.rsplit('/', 1)
    if len(parts) == 2 and parts[0]:
        _ensure_dir(parts[0])
    with open(path, 'w') as f:
        f.write(content)
    return {'ok': True, 'path': path}


@app.delete('/api/files')
async def delete_file(req):
    path = req.args.get('path', '')
    if not path:
        return {'error': 'path required'}, 400
    try:
        stat = os.stat(path)
        if stat[0] & 0x4000:
            os.rmdir(path)
        else:
            os.remove(path)
        return {'ok': True, 'path': path}
    except OSError as e:
        return {'error': 'Cannot delete {}: {}'.format(path, e)}, 404


@app.post('/api/files/mkdir')
async def mkdir(req):
    body = req.json
    path = body.get('path', '')
    if not path:
        return {'error': 'path required'}, 400
    _ensure_dir(path)
    return {'ok': True, 'path': path}


@app.post('/api/files/upload')
async def upload_file(req):
    path = req.args.get('path', '')
    if not path:
        return {'error': 'path required'}, 400
    parts = path.rsplit('/', 1)
    if len(parts) == 2 and parts[0]:
        _ensure_dir(parts[0])
    with open(path, 'wb') as f:
        f.write(req.body)
    return {'ok': True, 'path': path, 'size': len(req.body)}


def _ensure_dir(path):
    """Recursively create directories."""
    parts = path.split('/')
    current = ''
    for part in parts:
        if not part:
            continue
        current = '{}/{}'.format(current, part) if current else part
        try:
            os.mkdir(current)
        except:
            pass
```

```python
"""Config and WiFi API endpoints."""

import json
from lib.api.server import app
from lib.sys.log import log


@app.get('/api/config')
async def get_config(req):
    try:
        with open('config.json') as f:
            config = json.load(f)
    except:
        return {'error': 'config.json not found'}, 404

    # Redact API keys for security — show only last 4 chars
    providers = config.get('providers', {})
    for name, provider in providers.items():
        if isinstance(provider, dict) and 'api_key' in provider:
            key = provider['api_key']
            if len(key) > 4:
                provider['api_key'] = '***{}'.format(key[-4:])
            else:
                provider['api_key'] = '***'

    # Redact bot token
    tg = config.get('channels', {}).get('telegram', {})
    if 'bot_token' in tg:
        token = tg['bot_token']
        if len(token) > 4:
            tg['bot_token'] = '***{}'.format(token[-4:])
        else:
            tg['bot_token'] = '***'

    return config


@app.put('/api/config')
async def update_config(req):
    body = req.json
    if not isinstance(body, dict):
        return {'error': 'config must be a JSON object'}, 400

    # Validate required structure
    if 'providers' not in body:
        return {'error': 'config must contain providers'}, 400

    with open('config.json', 'w') as f:
        json.dump(body, f)

    log('info', 'API: config updated', source='api')
    return {'ok': True}


@app.get('/api/wifi')
async def get_wifi(req):
    result = {'ssid': None, 'connected': False, 'ip': None, 'rssi': None}
    try:
        from lib.wifi import get_credentials, is_connected, get_ip
        ssid, _ = get_credentials()
        result['ssid'] = ssid
        result['connected'] = is_connected()
        result['ip'] = get_ip()
        if is_connected():
            try:
                import network
                wlan = network.WLAN(network.STA_IF)
                result['rssi'] = wlan.status('rssi')
            except:
                pass
    except:
        pass
    return result


@app.put('/api/wifi')
async def set_wifi(req):
    body = req.json
    ssid = body.get('ssid', '')
    password = body.get('password', '')
    if not ssid:
        return {'error': 'ssid required'}, 400

    try:
        from lib.wifi import set_credentials, connect
        set_credentials(ssid, password)
        log('info', 'API: WiFi credentials updated', source='api')

        connected = connect(ssid, password, timeout=15)
        if connected:
            from lib.wifi import get_ip
            return {'ok': True, 'connected': True, 'ip': get_ip()}
        return {'ok': True, 'connected': False}
    except Exception as e:
        return {'error': 'WiFi setup failed: {}'.format(e)}, 500
```

```python
import gc
from lib.microdot.websocket import with_websocket
from lib.api.server import app

try:

@app.route('/ws/stats')
@with_websocket
async def stats_stream(req, ws):
    """Push system stats every 3 seconds."""
    try:
        while True:
            gc.collect()
            stats = {}

            try:
                free = gc.mem_free()
                alloc = gc.mem_alloc()
                stats['memory'] = {
                    'free_kb': free // 1024,
                    'used_kb': alloc // 1024,

                }
            except:
                stats['memory'] = None

            try:

                stats['temperature_c'] = esp32.mcu_temperature()
            except:
                stats['temperature_c'] = None

            try:
                from lib.wifi import is_connected, get_ip
                import network
                wlan = network.WLAN(network.STA_IF)
                stats['wifi'] = {
                    'connected': is_connected(),
                    'ip': get_ip(),
                    'rssi': wlan.status('rssi') if is_connected() else None,
                }
            except:
                stats['wifi'] = None

            try:
                import os
                stat = os.statvfs('/')
                block_size = stat[0]
                total_blocks = stat[2]
                free_blocks = stat[3]
                stats['storage'] = {
                    'total_kb': (block_size * total_blocks) // 1024,
                    'free_kb': (block_size * free_blocks) // 1024,
                }
            except:
                stats['storage'] = None

            try:
                import time
                stats['uptime_s'] = time.ticks_ms() // 1000
            except:
                try:
                    import time
                    stats['uptime_s'] = int(time.time())
                except:
                    stats['uptime_s'] = None

            import json
            await ws.send(json.dumps(stats))
            await asyncio.sleep(3)
    except:
        pass  # Client disconnected


@app.route('/ws/logs')
@with_websocket
async def logs_stream(req, ws):
    """Stream log messages in real time."""
    from lib.sys.log import subscribe, unsubscribe

    queue = []

    def _on_log(level, msg, source):
        queue.append({'level': level, 'msg': msg, 'source': source})

    subscribe(_on_log)
    try:
        import json
        while True:
            if queue:
                batch = queue[:]
                queue.clear()
                for entry in batch:
                    await ws.send(json.dumps(entry))
            await asyncio.sleep(0.5)
    except:
        pass
    finally:
        unsubscribe(_on_log)
```

```typescript
import type { ConnectionState, DeviceStatus, FileEntry } from '~/types/connection'

const state = reactive<ConnectionState>({
  mode: 'disconnected',
  serialConnected: false,
  networkConnected: false,
  deviceIp: null,
  devicePort: 8443,
  lastStatus: null,
  error: null,
})

let statsWs: WebSocket | null = null

function updateMode() {
  if (state.serialConnected && state.networkConnected) {
    state.mode = 'both'
  } else if (state.serialConnected) {
    state.mode = 'serial'
  } else if (state.networkConnected) {
    state.mode = 'network'
  } else {
    state.mode = 'disconnected'
  }
}

function baseUrl(): string {
  return `https://${state.deviceIp}:${state.devicePort}`
}

async function apiFetch<T>(path: string, options?: RequestInit): Promise<T> {
  const res = await fetch(`${baseUrl()}${path}`, options)
  if (!res.ok) {
    const body = await res.json().catch(() => ({ error: res.statusText }))
    throw new Error(body.error || res.statusText)
  }
  return res.json()
}

export function useConnection() {
  // --- Network ---

  async function connectNetwork(ip: string, port = 8443) {
    state.deviceIp = ip
    state.devicePort = port
    state.error = null
    try {
      const status = await apiFetch<any>('/api/status')
      state.networkConnected = true
      state.lastStatus = mapStatus(status)
      updateMode()
      startStatsStream()
    } catch (e: any) {
      state.error = e.message
      state.networkConnected = false
      updateMode()
    }
  }

  function disconnectNetwork() {
    stopStatsStream()
    state.networkConnected = false
    state.lastStatus = null
    updateMode()
  }

  // --- Stats WebSocket ---

  function startStatsStream() {
    stopStatsStream()
    const url = `wss://${state.deviceIp}:${state.devicePort}/ws/stats`
    statsWs = new WebSocket(url)
    statsWs.onmessage = (event) => {
      try {
        const data = JSON.parse(event.data)
        state.lastStatus = mapStatus(data)
      } catch {}
    }
    statsWs.onclose = () => {
      statsWs = null
    }
  }

  function stopStatsStream() {
    if (statsWs) {
      statsWs.close()
      statsWs = null
    }
  }

  // --- File operations ---

  async function listDir(path: string): Promise<FileEntry[]> {
    const res = await apiFetch<{ entries: any[] }>(`/api/files?path=${encodeURIComponent(path)}`)
    return res.entries.map(e => ({
      name: e.name,
      path: e.path,
      isDir: e.is_dir,
      size: e.size,
    }))
  }

  async function readFile(path: string): Promise<string> {
    const res = await apiFetch<{ content: string }>(`/api/files/read?path=${encodeURIComponent(path)}`)
    return res.content
  }

  async function writeFile(path: string, content: string): Promise<void> {
    await apiFetch('/api/files/write', {
      method: 'PUT',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ path, content }),
    })
  }

  async function deleteFile(path: string): Promise<void> {
    await apiFetch(`/api/files?path=${encodeURIComponent(path)}`, { method: 'DELETE' })
  }

  async function createDir(path: string): Promise<void> {
    await apiFetch('/api/files/mkdir', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ path }),
    })
  }

  async function uploadFile(path: string, data: ArrayBuffer): Promise<void> {
    await apiFetch(`/api/files/upload?path=${encodeURIComponent(path)}`, {
      method: 'POST',
      body: data,
    })
  }

  // --- Config ---

  async function getConfig(): Promise<Record<string, any>> {
    return apiFetch('/api/config')
  }

  async function saveConfig(config: Record<string, any>): Promise<void> {
    await apiFetch('/api/config', {
      method: 'PUT',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(config),
    })
  }

  // --- WiFi ---

  async function getWifi(): Promise<{ ssid: string | null, connected: boolean, ip: string | null, rssi: number | null }> {
    return apiFetch('/api/wifi')
  }

  async function setWifi(ssid: string, password: string): Promise<{ ok: boolean, connected: boolean, ip?: string }> {
    return apiFetch('/api/wifi', {
      method: 'PUT',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ ssid, password }),
    })
  }

  // --- Device control ---

  async function restartDevice(): Promise<void> {
    await apiFetch('/api/restart', { method: 'POST' })
    state.networkConnected = false
    updateMode()
  }

  return {
    state: readonly(state),
    connectNetwork,
    disconnectNetwork,
    listDir,
    readFile,
    writeFile,
    deleteFile,
    createDir,
    uploadFile,
    getConfig,
    saveConfig,
    getWifi,
    setWifi,
    restartDevice,
  }
}

function mapStatus(raw: any): DeviceStatus {
  return {
    agentName: raw.agent_name || raw.agentName || 'ZenClaw',
    version: raw.version || 'unknown',
    memory: raw.memory ? {
      freeKb: raw.memory.free_kb,
      usedKb: raw.memory.used_kb,
      totalKb: raw.memory.total_kb,
    } : null,
    temperatureC: raw.temperature_c ?? raw.temperatureC ?? null,
    wifi: raw.wifi ? {
      connected: raw.wifi.connected,
      ip: raw.wifi.ip,
      rssi: raw.wifi.rssi,
    } : null,
    storage: raw.storage ? {
      totalKb: raw.storage.total_kb,
      freeKb: raw.storage.free_kb,
    } : null,
    uptimeS: raw.uptime_s ?? raw.uptimeS ?? null,
  }
}
```

```vue
<script setup lang="ts">
const { state } = useConnection()

const modeLabel = computed(() => {
  switch (state.mode) {
    case 'serial': return 'USB'
    case 'network': return 'Network'
    case 'both': return 'USB + Network'
    default: return 'Disconnected'
  }
})

const modeColor = computed(() => {
  return state.mode === 'disconnected' ? 'neutral' : 'success'
})

const memoryPct = computed(() => {
  if (!state.lastStatus?.memory) return null
  const { usedKb, totalKb } = state.lastStatus.memory
  return Math.round((usedKb / totalKb) * 100)
})

function formatUptime(seconds: number | null): string {
  if (seconds == null) return '--'
  const h = Math.floor(seconds / 3600)
  const m = Math.floor((seconds % 3600) / 60)
  const s = seconds % 60
  return `${h}h ${m}m ${s}s`
}
</script>
```

```vue
<script setup lang="ts">
import type { FileEntry } from '~/types/connection'

const { state, listDir, readFile, writeFile, deleteFile, createDir, uploadFile } = useConnection()

const currentPath = ref('/')
const entries = ref<FileEntry[]>([])
const loading = ref(false)
const error = ref<string | null>(null)

// Editor state
const editingFile = ref<string | null>(null)
const editorContent = ref('')
const saving = ref(false)
const dirty = ref(false)

// New file/dir dialog
const showNewDialog = ref(false)
const newName = ref('')
const newIsDir = ref(false)

async function navigate(path: string) {
  loading.value = true
  error.value = null
  try {
    entries.value = await listDir(path)
    currentPath.value = path
    editingFile.value = null
  } catch (e: any) {
    error.value = e.message
  }
  loading.value = false
}

async function openFile(entry: FileEntry) {
  if (entry.isDir) {
    await navigate(entry.path)
    return
  }
  loading.value = true
  try {
    editorContent.value = await readFile(entry.path)
    editingFile.value = entry.path
    dirty.value = false
  } catch (e: any) {
    error.value = e.message
  }
  loading.value = false
}

async function save() {
  if (!editingFile.value) return
  saving.value = true
  try {
    await writeFile(editingFile.value, editorContent.value)
    dirty.value = false
  } catch (e: any) {
    error.value = e.message
  }
  saving.value = false
}

async function remove(entry: FileEntry) {
  if (!confirm(`Delete ${entry.name}?`)) return
  try {
    await deleteFile(entry.path)
    await navigate(currentPath.value)
  } catch (e: any) {
    error.value = e.message
  }
}

async function createNew() {
  const path = `${currentPath.value === '/' ? '' : currentPath.value}/${newName.value}`
  try {
    if (newIsDir.value) {
      await createDir(path)
    } else {
      await writeFile(path, '')
    }
    showNewDialog.value = false
    newName.value = ''
    await navigate(currentPath.value)
  } catch (e: any) {
    error.value = e.message
  }
}

async function handleUpload(event: Event) {
  const input = event.target as HTMLInputElement
  const file = input.files?.[0]
  if (!file) return
  const buffer = await file.arrayBuffer()
  const path = `${currentPath.value === '/' ? '' : currentPath.value}/${file.name}`
  try {
    await uploadFile(path, buffer)
    await navigate(currentPath.value)
  } catch (e: any) {
    error.value = e.message
  }
  input.value = ''
}

function goUp() {
  const parts = currentPath.value.split('/').filter(Boolean)
  parts.pop()
  navigate('/' + parts.join('/'))
}

function extension(name: string): string {
  const i = name.lastIndexOf('.')
  return i > 0 ? name.slice(i + 1) : ''
}

// Auto-navigate on mount
onMounted(() => navigate('/'))
</script>

<template>
  <div class="space-y-4">
    <div class="flex items-center justify-between">
      <h2 class="text-2xl font-bold text-white">Files</h2>
      <div class="flex gap-2">
        <UButton size="sm" variant="outline" icon="i-lucide-plus" @click="showNewDialog = true">
          New
        </UButton>
        <label>
          <UButton as="span" size="sm" variant="outline" icon="i-lucide-upload">
            Upload
          </UButton>
          <input type="file" class="hidden" @change="handleUpload">
        </label>
      </div>
    </div>

    <p v-if="!state.networkConnected" class="text-gray-400">
      Connect to a device first from the Dashboard.
    </p>

    <div v-else class="flex gap-4 min-h-[600px]">
      <!-- File list -->
      <div class="w-80 shrink-0">
        <div class="flex items-center gap-2 mb-2 text-sm text-gray-400">
          <UButton
            v-if="currentPath !== '/'"
            size="xs"
            variant="ghost"
            icon="i-lucide-arrow-up"
            @click="goUp"
          />
          <span class="font-mono truncate">{{ currentPath }}</span>
        </div>

        <div v-if="loading && !editingFile" class="py-8 text-center text-gray-500">
          Loading...
        </div>

        <div v-else class="space-y-0.5">
          <div
            v-for="entry in entries"
            :key="entry.path"
            class="flex items-center justify-between px-2 py-1.5 rounded hover:bg-gray-800 cursor-pointer group"
            :class="{ 'bg-gray-800': editingFile === entry.path }"
            @click="openFile(entry)"
          >
            <div class="flex items-center gap-2 min-w-0">
              <UIcon
                :name="entry.isDir ? 'i-lucide-folder' : 'i-lucide-file'"
                :class="entry.isDir ? 'text-blue-400' : 'text-gray-500'"
              />
              <span class="truncate text-sm text-gray-200">{{ entry.name }}</span>
            </div>
            <div class="flex items-center gap-2">
              <span v-if="entry.size != null" class="text-xs text-gray-600">
                {{ entry.size }}B
              </span>
              <UButton
                size="xs"
                variant="ghost"
                color="red"
                icon="i-lucide-trash-2"
                class="opacity-0 group-hover:opacity-100"
                @click.stop="remove(entry)"
              />
            </div>
          </div>
        </div>
      </div>

      <!-- Editor -->
      <div class="flex-1 flex flex-col">
        <div v-if="editingFile" class="flex-1 flex flex-col">
          <div class="flex items-center justify-between mb-2">
            <span class="font-mono text-sm text-gray-400">{{ editingFile }}</span>
            <UButton
```

```typescript
export interface FlashProgress {
  stage: 'connecting' | 'erasing' | 'flashing_firmware' | 'flashing_filesystem' | 'done' | 'error'
  percent: number
  message: string
}

export function useSerial() {
  const port = ref<SerialPort | null>(null)
  const connected = ref(false)

  async function requestPort(): Promise<boolean> {
    try {
      port.value = await navigator.serial.requestPort({
        filters: [{ usbVendorId: 0x303a }], // Espressif USB VID
      })
      return true
    } catch {
      return false
    }
  }

  async function openPort(baudRate = 115200): Promise<boolean> {
    if (!port.value) return false
    try {
      await port.value.open({ baudRate })
      connected.value = true
      return true
    } catch (e) {
      console.error('Failed to open serial port:', e)
      return false
    }
  }

  async function closePort() {
    if (port.value?.readable) {
      try {
        await port.value.close()
      } catch {}
    }
    connected.value = false
  }

  async function sendCommand(cmd: Record<string, any>): Promise<Record<string, any>> {
    if (!port.value?.writable || !port.value?.readable) {
      throw new Error('Serial port not open')
    }

    const writer = port.value.writable.getWriter()
    const encoder = new TextEncoder()
    const line = JSON.stringify(cmd) + '\n'
    await writer.write(encoder.encode(line))
    writer.releaseLock()

    // Read response line
    const reader = port.value.readable.getReader()
    const decoder = new TextDecoder()
    let buffer = ''
    try {
      while (true) {
        const { value, done } = await reader.read()
        if (done) break
        buffer += decoder.decode(value, { stream: true })
        const nlIndex = buffer.indexOf('\n')
        if (nlIndex >= 0) {
          const responseLine = buffer.slice(0, nlIndex).trim()
          return JSON.parse(responseLine)
        }
      }
    } finally {
      reader.releaseLock()
    }
    throw new Error('No response from device')
  }

  async function flashDevice(
    onProgress: (progress: FlashProgress) => void,
  ): Promise<boolean> {
    // Dynamic import of esptool-js
    const { ESPLoader, Transport } = await import('esptool-js')

    if (!port.value) {
      onProgress({ stage: 'error', percent: 0, message: 'No serial port selected' })
      return false
    }

    onProgress({ stage: 'connecting', percent: 0, message: 'Connecting to ESP32...' })

    try {
      const transport = new Transport(port.value)
      const loader = new ESPLoader({
        transport,
        baudrate: 921600,
        romBaudrate: 115200,
      })

      await loader.main()
      onProgress({ stage: 'connecting', percent: 10, message: `Chip: ${loader.chipName}` })

      // Fetch firmware files
      onProgress({ stage: 'flashing_firmware', percent: 20, message: 'Downloading firmware...' })
      const fwResponse = await fetch('/firmware/micropython.bin')
      const fwData = await fwResponse.arrayBuffer()

      onProgress({ stage: 'flashing_firmware', percent: 30, message: 'Flashing MicroPython firmware...' })
      await loader.writeFlash({
        fileArray: [{ data: new Uint8Array(fwData), address: 0x0 }],
        flashSize: 'keep',
        compress: true,
        reportProgress: (_fileIndex: number, written: number, total: number) => {
          const pct = 30 + Math.round((written / total) * 30)
          onProgress({ stage: 'flashing_firmware', percent: pct, message: `Firmware: ${written}/${total} bytes` })
        },
      })

      // Flash filesystem image
      onProgress({ stage: 'flashing_filesystem', percent: 65, message: 'Downloading ZenClaw filesystem...' })
      const fsResponse = await fetch('/firmware/zenclaw.img')
      const fsData = await fsResponse.arrayBuffer()

      onProgress({ stage: 'flashing_filesystem', percent: 70, message: 'Flashing ZenClaw files...' })
      // Filesystem partition offset depends on partition table — typically 0x310000 for default ESP32-S3
      const fsOffset = 0x310000
      await loader.writeFlash({
        fileArray: [{ data: new Uint8Array(fsData), address: fsOffset }],
        flashSize: 'keep',
        compress: true,
        reportProgress: (_fileIndex: number, written: number, total: number) => {
          const pct = 70 + Math.round((written / total) * 25)
          onProgress({ stage: 'flashing_filesystem', percent: pct, message: `Filesystem: ${written}/${total} bytes` })
        },
      })

      onProgress({ stage: 'done', percent: 100, message: 'Flash complete! Resetting device...' })
      await loader.hardReset()

      return true
    } catch (e: any) {
      onProgress({ stage: 'error', percent: 0, message: e.message })
      return false
    }
  }

  return {
    port: readonly(port),
    connected: readonly(connected),
    requestPort,
    openPort,
    closePort,
    sendCommand,
    flashDevice,
  }
}
```

```bash
# Build the ZenClaw filesystem image for flashing to ESP32-S3.
# Requires: mklittlefs (install via: pip install mklittlefs or build from source)
#
# Usage: ./scripts/build-firmware-image.sh
# Output: web/public/firmware/zenclaw.img

set -euo pipefail

SCRIPT_DIR="$(cd "$(dirname "$0")" && pwd)"
REPO_ROOT="$(dirname "$SCRIPT_DIR")"
FIRMWARE_DIR="$REPO_ROOT/firmware"
OUTPUT_DIR="$REPO_ROOT/web/public/firmware"
OUTPUT_FILE="$OUTPUT_DIR/zenclaw.img"

# ESP32-S3 default partition table: filesystem partition is typically 1.5MB
# Adjust this to match your partition table
PARTITION_SIZE=$((1536 * 1024))  # 1.5MB in bytes
BLOCK_SIZE=4096
PAGE_SIZE=256

echo "Building ZenClaw filesystem image..."
echo "  Source: $FIRMWARE_DIR"
echo "  Output: $OUTPUT_FILE"
echo "  Size:   $((PARTITION_SIZE / 1024))KB"

mkdir -p "$OUTPUT_DIR"

mklittlefs \
  -c "$FIRMWARE_DIR" \
  -s "$PARTITION_SIZE" \
  -b "$BLOCK_SIZE" \
  -p "$PAGE_SIZE" \
  "$OUTPUT_FILE"

echo "Done: $(du -h "$OUTPUT_FILE" | cut -f1) -> $OUTPUT_FILE"
```
